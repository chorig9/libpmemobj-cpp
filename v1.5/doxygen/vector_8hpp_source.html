513 template <typename T>
514 void
515 vector<T>::assign(size_type count, const_reference value)
516 {
517  pool_base pb = get_pool();
518 
519  transaction::run(pb, [&] {
520  if (count <= capacity()) {
521  /*
522  * Reallocation is not needed. First, replace old
523  * elements with new ones in range [0, size()).
524  * Depending on count, either call remaining old
525  * elements destructors, or append more new elements.
526  */
527  size_type size_old = _size;
528  snapshot_data(0, size_old);
529 
530  std::fill_n(
531  &_data[0],
532  (std::min)(count,
533  static_cast<size_type>(size_old)),
534  value);
535 
536  if (count > size_old) {
537 #if LIBPMEMOBJ_CPP_VG_PMEMCHECK_ENABLED
538  /*
539  * Range of memory:
540  * [&_data[size_old], &_data[count])
541  * is undefined, there is no need to snapshot
542  * and eventually rollback old data.
543  */
544  VALGRIND_PMC_ADD_TO_TX(
545  &_data[size_old],
546  sizeof(T) * (count - size_old));
547 #endif
548 
549  construct(size_old, count - size_old, value);
550  /*
551  * XXX: explicit persist is required here
552  * because given range wasn't snapshotted and
553  * won't be persisted automatically on tx
554  * commit. This can be changed once we will have
555  * implemented "uninitialized" flag for
556  * pmemobj_tx_xadd in libpmemobj.
557  */
558  pb.persist(&_data[size_old],
559  sizeof(T) * (count - size_old));
560  } else {
561  shrink(count);
562  }
563  } else {
564  dealloc();
565  alloc(count);
566  construct(0, count, value);
567  }
568  });
569 }

589 template <typename T>
590 template <typename InputIt,
591  typename std::enable_if<detail::is_input_iterator<InputIt>::value,
592  InputIt>::type *>
593 void
594 vector<T>::assign(InputIt first, InputIt last)
595 {
596  pool_base pb = get_pool();
597 
598  size_type size_new = static_cast<size_type>(std::distance(first, last));
599 
600  transaction::run(pb, [&] {
601  if (size_new <= capacity()) {
602  /*
603  * Reallocation is not needed. First, replace old
604  * elements with new ones in range [0, size()).
605  * Depending on size_new, either call remaining old
606  * elements destructors, or append more new elements.
607  */
608  size_type size_old = _size;
609  snapshot_data(0, size_old);
610 
611  InputIt mid = last;
612  bool growing = size_new > size_old;
613 
614  if (growing) {
615 #if LIBPMEMOBJ_CPP_VG_PMEMCHECK_ENABLED
616  /*
617  * Range of memory:
618  * [&_data[size_old], &_data[size_new])
619  * is undefined, there is no need to snapshot
620  * and eventually rollback old data.
621  */
622  VALGRIND_PMC_ADD_TO_TX(
623  &_data[size_old],
624  sizeof(T) * (size_new - size_old));
625 #endif
626 
627  mid = first;
628  std::advance(mid, size_old);
629  }
630 
631  iterator shrink_to = std::copy(first, mid, &_data[0]);
632 
633  if (growing) {
634  construct_range_copy(size_old, mid, last);
635  /*
636  * XXX: explicit persist is required here
637  * because given range wasn't snapshotted and
638  * won't be persisted automatically on tx
639  * commit. This can be changed once we will have
640  * implemented "uninitialized" flag for
641  * pmemobj_tx_xadd in libpmemobj.
642  */
643  pb.persist(&_data[size_old],
644  sizeof(T) * (size_new - size_old));
645  } else {
646  shrink(static_cast<size_type>(std::distance(
647  iterator(&_data[0]), shrink_to)));
648  }
649  } else {
650  dealloc();
651  alloc(size_new);
652  construct_range_copy(0, first, last);
653  }
654  });
655 }

1583  * which we will not overwrite
